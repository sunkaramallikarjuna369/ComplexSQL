The WHERE clause filters rows based on specified conditions. Only rows that satisfy the condition(s) are included in the result set. WHERE is used with SELECT, UPDATE, and DELETE statements.

ðŸ”§ HOW to use WHERE?
WHERE - All RDBMS Universal 
-- Basic comparison
SELECT * FROM employees WHERE salary > 50000;
SELECT * FROM employees WHERE department_id = 10;
SELECT * FROM employees WHERE first_name = 'John';

-- Multiple conditions with AND
SELECT * FROM employees 
WHERE department_id = 10 
  AND salary > 50000;

-- Multiple conditions with OR
SELECT * FROM employees 
WHERE department_id = 10 
   OR department_id = 20;

-- IN operator (cleaner than multiple OR)
SELECT * FROM employees 
WHERE department_id IN (10, 20, 30);

-- NOT IN
SELECT * FROM employees 
WHERE department_id NOT IN (10, 20);

-- BETWEEN (inclusive)
SELECT * FROM employees 
WHERE salary BETWEEN 50000 AND 100000;

-- LIKE pattern matching
SELECT * FROM employees WHERE first_name LIKE 'J%';     -- Starts with J
SELECT * FROM employees WHERE first_name LIKE '%son';   -- Ends with son
SELECT * FROM employees WHERE first_name LIKE '%oh%';  -- Contains oh
SELECT * FROM employees WHERE first_name LIKE 'J_hn';  -- _ = single char

-- IS NULL / IS NOT NULL
SELECT * FROM employees WHERE manager_id IS NULL;
SELECT * FROM employees WHERE commission IS NOT NULL;

-- Combining conditions with parentheses
SELECT * FROM employees 
WHERE (department_id = 10 OR department_id = 20)
  AND salary > 50000;

-- NOT operator
SELECT * FROM employees 
WHERE NOT (department_id = 10 AND salary < 50000);

-- Date comparisons
SELECT * FROM employees 
WHERE hire_date >= '2020-01-01';

SELECT * FROM employees 
WHERE hire_date BETWEEN '2020-01-01' AND '2020-12-31';

Performance Tips:
  • Use indexed columns in WHERE clauses
  • Avoid functions on indexed columns (e.g., WHERE YEAR(date) = 2024)
  • Use IN instead of multiple OR conditions
  • Be careful with NULL comparisons (use IS NULL, not = NULL)
