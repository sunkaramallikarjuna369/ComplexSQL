üìò WHAT is DML?

DML (Data Manipulation Language) consists of SQL commands used to retrieve, insert, modify, and delete data in database tables. Unlike DDL, DML operations work with the data itself, not the structure.

üéØ WHY use DML?

DML commands are the core of database interaction:

  • Data Retrieval: SELECT is the most used SQL command for querying data
  • Data Entry: INSERT allows adding new records to tables
  • Data Maintenance: UPDATE keeps data current and accurate
  • Data Cleanup: DELETE removes obsolete or incorrect data
  • Transaction Support: DML operations can be rolled back within transactions
üí°
Pro Tip: Always use WHERE clause with UPDATE and DELETE to avoid accidentally modifying or removing all rows. Test your WHERE clause with SELECT first!

1. INSERT Statement

INSERT - SQL Server SQL Server 
-- Single row insert
INSERT INTO employees (first_name, last_name, email, salary)
VALUES ('John', 'Doe', 'john@company.com', 75000);

-- Multiple rows insert
INSERT INTO employees (first_name, last_name, email, salary)
VALUES 
    ('Jane', 'Smith', 'jane@company.com', 80000),
    ('Bob', 'Johnson', 'bob@company.com', 65000),
    ('Alice', 'Williams', 'alice@company.com', 90000);

-- Insert from SELECT
INSERT INTO employees_archive (first_name, last_name, email)
SELECT first_name, last_name, email
FROM employees
WHERE hire_date < '2020-01-01';

-- Insert with OUTPUT (return inserted values)
INSERT INTO employees (first_name, last_name, salary)
OUTPUT inserted.employee_id, inserted.first_name
VALUES ('New', 'Employee', 70000);

INSERT INTO employees (employee_id, first_name, last_name, salary)
VALUES (1, 'John', 'Doe', 80000)
ON DUPLICATE KEY UPDATE salary = VALUES(salary);

INSERT IGNORE INTO employees (employee_id, first_name, last_name)
VALUES (1, 'John', 'Doe');

2. UPDATE Statement

UPDATE - SQL Server SQL Server 
-- Simple update
UPDATE employees
SET salary = 80000
WHERE employee_id = 1;

-- Update multiple columns
UPDATE employees
SET salary = salary * 1.10,
    last_name = 'Smith-Jones'
WHERE department_id = 1;

-- Update with JOIN
UPDATE e
SET e.salary = e.salary * 1.15
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
WHERE d.department_name = 'Engineering';

-- Update with OUTPUT
UPDATE employees
SET salary = salary + 5000
OUTPUT deleted.salary AS old_salary, 
       inserted.salary AS new_salary
WHERE employee_id = 1;
UPDATE - Oracle Oracle 
-- Simple update
UPDATE employees
SET salary = 80000
WHERE employee_id = 1;

-- Update multiple columns
UPDATE employees
SET salary = salary * 1.10,
    last_name = 'Smith-Jones'
WHERE department_id = 1;

-- Update with subquery
UPDATE employees e
SET salary = salary * 1.15
WHERE EXISTS (
    SELECT 1 FROM departments d
    WHERE d.department_id = e.department_id
    AND d.department_name = 'Engineering'
);

-- Update with RETURNING
UPDATE employees
SET salary = salary + 5000
WHERE employee_id = 1
RETURNING salary INTO v_new_salary;

3. DELETE Statement

DELETE - All RDBMS Universal 
-- Simple delete
DELETE FROM employees
WHERE employee_id = 1;

-- Delete with multiple conditions
DELETE FROM employees
WHERE department_id = 5
AND hire_date < '2020-01-01';

-- Delete all rows (use with caution!)
DELETE FROM employees;

-- Delete with subquery
DELETE FROM employees
WHERE department_id IN (
    SELECT department_id 
    FROM departments 
    WHERE location = 'Closed Office'
);

Safety First: Always test your WHERE clause with SELECT before running UPDATE or DELETE: 
-- First, verify which rows will be affected
SELECT * FROM employees WHERE department_id = 5;

-- Then, if correct, run the DELETE
DELETE FROM employees WHERE department_id = 5;
