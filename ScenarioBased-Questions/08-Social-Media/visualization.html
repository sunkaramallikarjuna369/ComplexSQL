Q141 Medium CTE Trending Analysis

Scenario: Discover what's popular right now

Find trending hashtags

WITH recent_hashtags AS (
    SELECT h.hashtag_id, h.tag_name, COUNT(*) AS recent_posts
    FROM hashtags h
    JOIN post_hashtags ph ON h.hashtag_id = ph.hashtag_id
    JOIN posts p ON ph.post_id = p.post_id
    WHERE p.created_at >= CURRENT_TIMESTAMP - INTERVAL '24 hours'
    GROUP BY h.hashtag_id, h.tag_name
)
SELECT tag_name, recent_posts,
       ROUND(100.0 * (recent_posts - prev_posts) / NULLIF(prev_posts, 0), 2) AS growth_pct
FROM recent_hashtags ORDER BY recent_posts DESC LIMIT 10;

Q142 Medium Engagement Influencer Analytics

Scenario: Influencer analytics

Calculate user engagement rate

SELECT u.user_id, u.username, u.follower_count,
       COUNT(DISTINCT p.post_id) AS total_posts,
       SUM(p.likes_count) AS total_likes,
       SUM(p.comments_count) AS total_comments,
       ROUND(100.0 * (SUM(p.likes_count) + SUM(p.comments_count)) / 
             (NULLIF(u.follower_count, 0) * COUNT(DISTINCT p.post_id)), 4) AS engagement_rate
FROM users u
JOIN posts p ON u.user_id = p.user_id
WHERE p.created_at >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY u.user_id, u.username, u.follower_count
ORDER BY engagement_rate DESC;

Q143 Hard Self Join Friend Suggestions

Scenario: Friend suggestions

Find mutual followers

SELECT f1.follower_id AS user_id, u1.username AS user_name,
       f2.following_id AS suggested_user_id, u2.username AS suggested_name,
       COUNT(*) AS mutual_connections
FROM follows f1
JOIN follows f2 ON f1.following_id = f2.follower_id
JOIN users u1 ON f1.follower_id = u1.user_id
JOIN users u2 ON f2.following_id = u2.user_id
WHERE f1.follower_id != f2.following_id
AND NOT EXISTS (SELECT 1 FROM follows f3 
                WHERE f3.follower_id = f1.follower_id AND f3.following_id = f2.following_id)
GROUP BY f1.follower_id, u1.username, f2.following_id, u2.username
HAVING COUNT(*) >= 3;
Q145 Hard Bot Detection Platform Integrity

Scenario: Platform integrity

Detect fake/bot accounts

SELECT u.user_id, u.username, u.follower_count, u.following_count,
       COUNT(DISTINCT p.post_id) AS post_count,
       ROUND(u.following_count::DECIMAL / NULLIF(u.follower_count, 0), 2) AS following_ratio,
       CASE WHEN u.following_count > 1000 AND u.follower_count < 100 THEN 'HIGH'
            WHEN COUNT(DISTINCT p.post_id) = 0 AND COUNT(DISTINCT l.like_id) > 100 THEN 'HIGH'
            ELSE 'LOW' END AS bot_risk
FROM users u
LEFT JOIN posts p ON u.user_id = p.user_id
LEFT JOIN likes l ON u.user_id = l.user_id
GROUP BY u.user_id, u.username, u.follower_count, u.following_count
HAVING u.following_count > 1000 AND u.follower_count < 100;
